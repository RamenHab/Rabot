//Счетчик дней.Сколько спортсмен пробежит за семь дней, увеличивая свой предыдущий рекорд 10км на 10% каждый день.
#include <iostream>
#include <math.h>
using namespace std;
int main()
{
    //float задает переменные
float i,a,n,s;
//Переменной а было присвоено значение 10
    a=10;
//Счетчик дней
    for (i=1; i<=7; i++)
    {
//Вычисление по формуле
    n=a+((a*10)/100);
    s=s+n;
    }
//Ответ Итог. Завершение программы.
    cout « s;
 return 0;
}
