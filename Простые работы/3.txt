#include <iostream>
#include<math.h>
using namespace std;
int main() 
{
    float s;
    cout << "\n Введите число = "; cin >> s;
    if (s==1) cout <<"\n Понедельник";
    if (s==2) cout <<"\n Вторник";
    if (s==3) cout <<"\n Среда";
    if (s==4) cout <<"\n Четверг";
    if (s==5) cout <<"\n Пятница";
    if (s==6) cout <<"\n Суббота";
    if (s==7) cout <<"\n Воскресенье";
    return 0;
}
